create selectdeputy song_2014_2022 as select name, singer, date from Song where date>=2014 and date<=2022;
select * from song_2014_2022;
create uniondeputy song_photo as select name, singer, date from Song union select name, singer, date from Photo;
select * from song_photo;
create joindeputy singer_song as select Song.name, Song.date, Song.singer, Singer.sex, Singer.age, Singer.nationality, Singer.company from Song, Singer where Song.singer = Singer.name;
select * from singer_song;
create groupdeputy year_songnumber as select date, count(id) from Song group by date;
select * from year_songnumber;
